function typingEffect(element,speed,timer){

    setTimeout(function()
    { 
        element.style.display = "flex";
        
        let text=element.innerHTML;

        element.innerHTML="";
        var i = 0;

        var timer = setInterval(function(){

        if(i < text.length){
            element.append(text.charAt(i))
            i++;
        }else{
            clearInterval(timer);
            }
        },speed)        
    },timer);
}

const h1=document.getElementById('hello');
typingEffect(h1,150, 0);

const h2=document.getElementById('naming');
typingEffect(h2,150, 1200);
